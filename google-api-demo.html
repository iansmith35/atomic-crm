GET /api/gmail/unread

Purpose: Retrieve unread emails from Gmail (max 20)

fetch('/api/gmail/unread') .then(res => res.json()) .then(emails => console.log(emails));

Sample Response:

[{ "id": "1", "subject": "Important Business Update", "from": "client@example.com", "date": "2024-01-15T10:30:00Z", "snippet": "This is an important update...", "isRead": false }]

POST /api/gmail/send

Purpose: Send an email through Gmail

fetch('/api/gmail/send', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ to: 'recipient@example.com', subject: 'Hello', body: 'Email content here' }) });

Sample Response:

{ "id": "sent_1234567890", "to": "recipient@example.com", "subject": "Hello", "sentAt": "2024-01-15T10:30:00Z", "success": true }

GET /api/calendar/events

Purpose: List calendar events with optional filtering

fetch('/api/calendar/events?calendarId=primary&timeMin=2024-01-15T00:00:00Z') .then(res => res.json()) .then(events => console.log(events));

Sample Response:

[{ "id": "event_1", "summary": "Team Meeting", "start": { "dateTime": "2024-01-15T10:00:00Z" }, "end": { "dateTime": "2024-01-15T11:00:00Z" }, "attendees": [{"email": "team@company.com"}] }]

POST /api/calendar/events

Purpose: Create a new calendar event

fetch('/api/calendar/events', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ event: { summary: 'New Meeting', start: { dateTime: '2024-01-15T10:00:00Z', timeZone: 'UTC' }, end: { dateTime: '2024-01-15T11:00:00Z', timeZone: 'UTC' } } }) });

Sample Response:

{ "id": "created_1234567890", "summary": "New Meeting", "status": "confirmed", "htmlLink": "https://calendar.google.com/event?eid=1234567890" }
